Header Right

Features​

  1. Shortcut Button:
  • Displays a customizable shortcut icon (pin_star or pin_star_checked by default).
  • Toggles the "favorite" state using the onPressShortcut handler.
  1. Help Center Button:
  • Launches the help center using onPressHelpCenter.
  1. Close Button:
  • Closes the current modal or screen with onClose.
  • If preventClose is provided, shows a confirmation popup before closing.
  1. Dynamic Tools:
  • Supports multiple tools that can be grouped.
  • Displays a "more" icon (navigation_more_icon) for accessing additional tools if there are multiple items.
  1. Badge Support:
  • Shows a badge on the shortcut button if any tool has showBadge: true.

Example Usage​

import React, {useEffect} from 'react';

const ExampleHeaderToolkit = ({navigation}) => {
  const tools = [
    {
      title: 'Group 1',
      items: [
        {
          key: 'tool1',
          icon: 'icon_tool1',
          onPress: () => console.log('Tool 1 pressed'),
        },
        {
          key: 'tool2',
          icon: 'icon_tool2',
          onPress: () => console.log('Tool 2 pressed'),
        },
      ],
    },
    {
      title: 'Group 2',
      items: [
        {
          key: 'tool3',
          icon: 'icon_tool3',
          onPress: () => console.log('Tool 3 pressed'),
          showBadge: true,
        },
      ],
    },
  ];

  useEffect(() => {
    navigation?.setOptions({
      headerRight: {
        useShortcut: true,
        tools
      },
    });
  }, []);
};

export default ExampleHeaderToolkit;

Shortcut​

  • Default it will be a "Pin to home" action.
  • Custom 1 action -> Show custom action
  • Custom 2 or more actions -> Show more icon
  • Prevent close to show popup ask for close
import React, {FC, useContext, useEffect, useState} from 'react';
import {
  ApplicationContext,
  Button,
  Card,
  Input,
  Item,
  NavigationScreenProps,
  PopupNotifyProps,
  Radio,
  Screen,
  Spacing,
  Text,
} from '@momo-kits/foundation';
import {Alert} from 'react-native';

const ToolkitUsage: FC<NavigationScreenProps> = ({navigation}) => {
  const testValue = ['onFavorite', 'test2', 'something'];
  const [badgeValue, setBadgeValue] = useState('');
  const [pin, setPin] = useState('');
  const [badgeType, setBadgeType] = useState(undefined);
  const {navigator} = useContext(ApplicationContext);
  const [useShortcut, setUseShortcut] = useState(true);
  useEffect(() => {
    const popup: PopupNotifyProps = {
      buttonDirection: 'row',
      primary: {
        title: 'Về Trang Trước',
        onPress: () => {},
      },
      secondary: {
        title: 'Ở lại',
        onPress: () => {},
      },
      description:
        'Cell team tùy ý custom nội dung text ở title, description và 2 nút CTA. Không custom bất kì style nào trong popup',
      title: 'Title',
    };

    navigation.setOptions({
      headerRight: {
        preventClose: popup,
      },
    });
  }, []);

  const iconSource =
    'https://img.mservice.com.vn/app/img/kits/32_notifications_bell_alert.png';

  return (
    <Screen navigation={navigation} scrollable={true}>
      <Card>
        <Item>
          <Input
            floatingValue={'Badge value'}
            placeholder={'Input badge value'}
            value={badgeValue}
            onChangeText={text => {
              setBadgeValue(text);
            }}
          />
        </Item>
        <Item>
          <Text
            style={{marginBottom: Spacing.M}}
            typography={'label_default_medium'}>
            Config pinned tool
          </Text>
          {testValue.map(item => {
            return (
              <Radio
                value={item}
                onChange={() => setPin(item)}
                groupValue={pin}
                label={item}
              />
            );
          })}
        </Item>
        <Item>
          <Text
            style={{marginBottom: Spacing.M}}
            typography={'label_default_medium'}>
            Badge type
          </Text>
          {['number', 'dot'].map(item => {
            return (
              <Radio
                value={item}
                onChange={() => setBadgeType(item)}
                groupValue={badgeType}
                label={item}
              />
            );
          })}
        </Item>
        <Item>
          <Button
            title={'Show/Hide shortcut'}
            onPress={() => {
              navigation?.setOptions({
                headerRight: {
                  useShortcut: !useShortcut,
                },
              });
              setUseShortcut(!useShortcut);
            }}
          />
        </Item>
        <Item>
          <Button
            title={'Custom 1 shortcut'}
            onPress={() => {
              navigation?.setOptions({
                headerRight: {
                  useShortcut: true,
                  tools: [
                    {
                      title: {vi: 'Title 1', en: 'Title 1'},
                      items: [
                        {
                          name: {
                            vi: 'TITLE 2',
                            en: 'EN',
                          },
                          key: 'test 2',
                          icon: iconSource,
                          showBadge: true,
                          onPress: () => {
                            Alert.alert('Test service 2');
                          },
                        },
                      ],
                    },
                  ],
                },
              });
            }}
          />
        </Item>
        <Item>
          <Button
            title={'Use more'}
            onPress={() => {
              navigation?.setOptions({
                headerRight: {
                  useShortcut: true,
                  useMore: true,
                  tools: [
                    {
                      title: {vi: 'Title 1', en: 'Title 1'},
                      items: [
                        {
                          name: {
                            vi: 'TITLE 2',
                            en: 'EN',
                          },
                          key: 'test 2',
                          icon: iconSource,
                          showBadge: true,
                          onPress: () => {
                            Alert.alert('Test service 2');
                          },
                        },
                      ],
                    },
                  ],
                },
              });
            }}
          />
        </Item>
        <Item>
          <Button
            title={'Custom 2 shortcut'}
            onPress={() => {
              navigation?.setOptions({
                headerRight: {
                  useShortcut: true,
                  tools: [
                    {
                      title: {vi: 'Title 1', en: 'Title 1'},
                      items: [
                        {
                          name: {
                            vi: 'TITLE 1',
                            en: 'EN',
                          },
                          key: 'test 1',
                          icon: iconSource,
                          showBadge: true,
                          onPress: () => {
                            Alert.alert('Test service 1');
                          },
                        },
                        {
                          name: {
                            vi: 'TITLE 2',
                            en: 'EN',
                          },
                          key: 'test 2',
                          icon: iconSource,
                          showBadge: true,
                          onPress: () => {
                            Alert.alert('Test service 2');
                          },
                        },
                      ],
                    },
                  ],
                },
              });
            }}
          />
        </Item>
      </Card>
    </Screen>
  );
};

export default ToolkitUsage;

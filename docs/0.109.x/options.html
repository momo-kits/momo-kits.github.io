  subTitle: 'Default subtitle',
  image:

  dotColor: Colors.green_03,
  verify: true,

    Alert.alert('You pressed');

const UserTitleUsage = ({navigation}) => {

  }, []);
  return <Screen></Screen>;

Concept

HeaderTypes​

  • HeaderType includes 3 types default, extended, none.
  • It also has other templates such as search header or search animated header.
  • inputSeachProps: Props of input search in type of animated search header.
  • headerRightWidth: To config width of headerRight component if you custom it.
  • useHeaderShadow: To config shadow in bottom of header (if false, it will be set as a border)
  • use ScreenRef to config searchHeader without animation (example below)
import {
  ApplicationContext,
  Button,
  Card,
  HeaderType,
  Input,
  Item,
  NavigationScreenProps,
  ScreenRef,
  Styles,
  Switch,
} from '@momo-kits/foundation';
import {useContext, useEffect, useState} from 'react';
import {View} from 'react-native';
const HeaderTypeUsage: React.FC<NavigationScreenProps> = ({navigation}) => {
  const {theme} = useContext(ApplicationContext);
  const screenRef = React.useRef<ScreenRef>();
  const [headerType, setHeaderType] = useState<HeaderType>('default');
  const [showSearchAnimated, setShowSearchAnimated] = useState(false);
  const [showSearchHeader, setShowSearchHeader] = useState(false);
  const [useHeaderShadow, setUseHeaderShadow] = useState(true);

  const data = [
    {
      value: 'default',
      props: {value: 'default', groupValue: 'default'},
    },
    {
      value: 'extended',
      props: {value: 'extended', groupValue: 'extended'},
    },
    {
      value: 'none',
      props: {value: 'none', groupValue: 'none'},
    },
  ];

  useEffect(() => {
    setTimeout(() => {
      screenRef.current?.setOptions({title: 'Hello'});
    }, 1000);
  }, []);

  return (
    <Screen
      ref={screenRef}
      inputSearchProps={
        showSearchAnimated
          ? {
              placeholder: 'Search input',
              leftPosition: 12,
            }
          : undefined
      }
      headerType={headerType}
      headerRightWidth={0}
      navigation={navigation}
      scrollable={true}
      backgroundColor={theme.colors.background.default}
      useShadowHeader={useHeaderShadow}>
      <Card>
        <Item>
          {data?.map(item => {
            return (
              <Radio
                label={item.value}
                value={headerType}
                onChange={() => setHeaderType(item.value as HeaderType)}
                groupValue={item?.props?.groupValue}
              />
            );
          })}
        </Item>
      </Card>
      <Card>
        <Item>
          <View style={Styles.rowSpace}>
            <Text typography={'title_default'}>Search Animated</Text>
            <Switch
              onChange={() => {
                const value = !showSearchAnimated;
                if (value) {
                  setShowSearchHeader(false);
                }
                setShowSearchAnimated(value);
              }}
              value={showSearchAnimated}
            />
          </View>
        </Item>
      </Card>
      <Card>
        <Item>
          <View style={Styles.rowSpace}>
            <Text typography={'title_default'}>Search Header</Text>
            <Switch
              onChange={() => {
                const value = !showSearchHeader;
                if (value) {
                  setShowSearchAnimated(false);
                }
                setShowSearchHeader(value);
              }}
              value={showSearchHeader}
            />
          </View>
          {showSearchHeader && (
            <>
              <Input
                onChangeText={(text: string) => {
                  screenRef.current?.setSearchHeader({
                    defaultValue: 'Search input',
                    onChangeText: text => {
                      console.log(text);
                    },
                    onPress: () => {
                      console.log('onPress');
                    },
                    buttonText: text,
                    showButtonText: true,
                  });
                }}
              />
              <Button
                size={'medium'}
                title={'Show search header'}
                onPress={() => {
                  screenRef.current?.setSearchHeader({
                    defaultValue: 'Search input',
                    onChangeText: text => {
                      console.log(text);
                    },
                    onPress: () => {
                      console.log('onPress');
                    },
                  });
                }}
              />
            </>
          )}
        </Item>
      </Card>
      <Card>
        <Item>
          <View style={Styles.rowSpace}>
            <Text typography={'title_default'}>use Header Shadow</Text>
            <Switch
              onChange={() => {
                setUseHeaderShadow(!useHeaderShadow);
              }}
              value={useHeaderShadow}
            />
          </View>
        </Item>
      </Card>
      <Card>
        <Item>
          <View style={{height: 1000}} />
        </Item>
      </Card>
    </Screen>
  );
};

AnimatedHeader​

import {
  ApplicationContext,
  Colors,
  HeaderAnimated,
  NavigationScreenProps,
  Screen,
  ScreenRef,
} from '@momo-kits/foundation';
import {RefreshControl} from 'react-native';
import React, {useContext, useRef, useState} from 'react';

export const AnimatedHeader: React.FC<NavigationScreenProps> = ({
  navigation,
}) => {
  const {theme} = useContext(ApplicationContext);
  const ref = useRef<ScreenRef>();
  const [refreshing, setRefresh] = useState(false);
  const [surface, setSurface] = useState(true);
  const [scale, setScale] = useState(true);

  return (
    <Screen
      ref={ref}
      scrollable
      navigation={navigation}
      layoutOffset={-56}
      scrollViewProps={{
        scrollEventThrottle: 8,
        keyboardShouldPersistTaps: 'handled',
        onScroll: () => {},
        refreshControl: (
          <RefreshControl
            refreshing={refreshing}
            tintColor={theme.colors.background.surface}
            onRefresh={() => {
              setRefresh(true);
              setTimeout(() => {
                setRefresh(false);
              }, 3000);
            }}
            colors={[
              theme.colors.background.default,
              theme.colors.background.surface,
            ]}
          />
        ),
      }}
      animatedHeader={{
        headerTintColor: surface ? Colors.black_01 : Colors.black_17,
        component: (props: any) => (
          <HeaderAnimated
            {...props}
            onPress={() => {}}
            useScale={scale}
            image={
              'https://static.momocdn.net/app/img/booking/holiday/mid_atumn_dldl.jpeg'
            }
          />
        ),
      }}
      enableKeyboardAvoidingView={true}></Screen>
  );
};

import {RefreshControl, ScrollView} from 'react-native';

import {HeaderSliderBanner} from '@momo-kits/template';

export const HeaderSliderBannerUsage: React.FC<NavigationScreenProps> = ({
  navigation,
}) => {

  const ref = useRef<ScrollView>();
  const [refreshing, setRefresh] = useState(false);
  const [activeIndex, setActiveIndex] = useState(0);
  const [surface, setSurface] = useState(false);

  return (
    <Screen
      ref={ref}
      scrollable
      layoutOffset={-56}
      navigation={navigation}
      scrollViewProps={{
        scrollEventThrottle: 8,

        onScroll: e => {
          console.log('onScroll', e);
        },

          <RefreshControl
            refreshing={refreshing}

};
